# 运算符重载——初探

标签（空格分隔）： C++

---

例子：
```cpp
#include<iostream>

using namespace std;

class Time
{
    public:
    Time(int h, int min)
    {
        this->h=h;
        this->min=min;
    }
    ~Time(){}
    int h;
    int min;
    
    //在类的内部进行运算符重载
    Time operator-(const Time &t)const
    {
        Time sub(0,0);
        sub.h = h - t.h;
        sub.min = min - t.min;
        return sub;
    }
}

int main(int argc, char* argv[])
{
    Time t1(1, 1);
    Time t2(2, 0);
    Time t = t1- t2;

    cout << t.h <<"  "<<t.min<<endl;//输出结果：-1  1
    system("PAUSE");
    return 0;
}

```
